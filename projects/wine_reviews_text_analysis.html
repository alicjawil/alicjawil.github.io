def missing_values(df):
            mis_val = df.isnull().sum()
            mis_val_percent = 100 * df.isnull().sum() / len(df)
            mis_val_table = pd.concat([mis_val, mis_val_percent], axis=1)
            mis_val_table_ren_columns = mis_val_table.rename(
            columns = {0 : 'Missing Values', 1 : '% of Total Values'})
            mis_val_table_ren_columns = mis_val_table_ren_columns[
                mis_val_table_ren_columns.iloc[:,1] != 0].sort_values(
            '% of Total Values', ascending=False).round(1)
            print ("Your selected dataframe has " + str(df.shape[1]) + " columns.\n"      
                "There are " + str(mis_val_table_ren_columns.shape[0]) +
                " columns that have missing values.")
            return mis_val_table_ren_columns

listCorpus = wine['description'].tolist()

tfidf_bigram = TfidfVectorizer(ngram_range = (2,2), stop_words = ENGLISH_STOP_WORDS)
tfidf_bigram.fit(listCorpus)
vector = tfidf_bigram.transform(wine['description'])
max_value = vector.max(axis=0).toarray().ravel()
sorted_by_tfidf = max_value.argsort()

# Bigram
#Not as valuable - low tfidf
feature_names = np.array(tfidf_bigram.get_feature_names())
print("\nFeatures with the lowest tfidf - Bigram :\n{}".format(feature_names[sorted_by_tfidf[:20]]))

#Frequently unique and valueable and most relevant, high tfidf
print("\nFeatures with the highest tfidf - Bigram :\n{}".format(feature_names[sorted_by_tfidf[-20:]]))

#idf score low, frequent term
sorted_by_idf = np.argsort(tfidf_bigram.idf_)
print("\nFeatures with the lowest idf - Bigram :\n{}".format(feature_names[sorted_by_idf[:20]]))

#highest idf, rare words
print("\nFeatures with the highest idf - Bigram :\n{}".format(feature_names[sorted_by_idf[-20:]]))
